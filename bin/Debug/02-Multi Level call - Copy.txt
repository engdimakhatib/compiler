{
     call multiply( input x,y output z);

procedure multiply( input x,y output z);
{
     call divi (input x,y output z);
     writeln(' z=  ', z);     
     z:= z + x*y;
}

procedure divi( input x,y output z);
{
      z := x / y;
}
